/// 标准用词：
/// 1. COC CANOPEN控制
/// 2. HWT HotWaterTank 水箱
/// 3. Rt RealTime 实时
/// 4. DBP DryBoilingPrevention 防干烧
/// 5. WPC WaterPreparationControl 备水控制, 表示缺水时自动补水的功能
/// 6. WTC WaterTemperatureControl 温度控制，表示通过控制加热棒实现将水温加热到目标温度（全功率加热输出），并保持在目标温度的功能（PID加热输出）
/// 7. HWDC HotWaterDoseControl 热水定量输出控制
/// 8. BTC BrewTeaControl 泡茶控制
/// 9. TBPC TeaBagPickControl 茶包控制
/// 10.HWTC HotWaterTankControl 热水箱控制
/// 11.NMT NetworkManagement 网络管理，表示对从机的控制和管理功能

/************ 通用定义 ************/

enum PkgType {
  READ_CFG_REQ = 0x01;            // 读取配置
  READ_CFG_RSP = 0x02;            // 读取配置响应
  WRITE_CFG_REQ = 0x03;           // 写入配置
  WRITE_CFG_RSP = 0x04;           // 写入配置响应
  EVENT_REPORT = 0x05;            // 主动上报事件
  DO_TASK_REQ = 0x06;             // 执行任务请求
  TASK_RCV_RSP = 0x07;            // 任务接收结果
  TASK_PROGRESS_RPT = 0x08;       // 任务进度报告
  TASK_EXC_RSP = 0x09;            // 任务执行结果
  TASK_CANCEL_REQ = 0x0A;         // 取消任务请求
  TASK_CANCEL_RSP = 0x0B;         // 任务取消应答
  EXC_CMD_REQ = 0x0C;             // 执行命令请求
  EXC_CMD_RSP = 0x0D;             // 执行命令应答
  SEND_DATA_REQ = 0x0E;           // 传输数据请求
  SEND_DATA_RSP = 0x0F;           // 传输数据应答
  QUERY_SLAVE_REQ = 0x10;         // 查询从机请求
  QUERY_SLAVE_RSP = 0x11;         // 查询从机回复
  QUERY_SLAVE_TASK_ID_REQ = 0x12; // 查询从机正在执行的任务ID请求
  QUERY_SLAVE_TASK_ID_RSP = 0x13; // 查询从机正在执行的任务ID响应
}

// 功能模组、电路板类型、ID
message FuncModuleId {
  required uint32 Function = 1 [ (nanopb).int_size = IS_8 ];    // 功能模组
  required uint32 FuncVersion = 2 [ (nanopb).int_size = IS_8 ]; // 功能模组版本
  required uint32 BoardType = 3 [ (nanopb).int_size = IS_8 ];   // 电路板类型
  required uint32 SlaveId = 4 [ (nanopb).int_size = IS_8 ];     // 单元Id
}
/************************************************ 同步指令报文定义 **********************************************/
/************************************************ 查询从机指令报文定义 *******************************************/
message FunctionList {
  required uint32 Function = 1 [ (nanopb).int_size = IS_8 ];
  required uint32 FuncVersion = 2 [ (nanopb).int_size = IS_8 ]; // 功能模组版本
}
// 从机数据
message SlaveData {
  repeated FunctionList FunctionList = 1 [ (nanopb).max_count = 10 ]; // 功能表
  optional SlaveFixedInfo SlaveFixedInfo = 2;                         // 从站信息
  optional SlaveChangeInfo SlaveChangeInfo = 3;                       // 从站信息
}
// 从机固定信息
message SlaveFixedInfo {
  required uint32 SlaveID = 1 [ (nanopb).int_size = IS_8 ];          // 从站ID
  required uint32 BoardType = 2 [ (nanopb).int_size = IS_8 ];        // 电路板类型
  required bool HardJumperSwitch = 3;                                // 硬件拨码开关 状态，true表示硬件拨码，false表示软件设置ID
  required uint32 UUID = 4 [ (nanopb).int_size = IS_32 ];            // 设备识别码
  required uint32 VendorID = 5 [ (nanopb).int_size = IS_32 ];        // VendorID
  required uint32 ProductID = 6 [ (nanopb).int_size = IS_32 ];       // ProductID
  required uint32 FirmwareVersion = 7 [ (nanopb).int_size = IS_32 ]; // 固件版本
  required uint32 HardwareVersion = 8 [ (nanopb).int_size = IS_32 ]; // 硬件版本
}

// 从机变化信息
message SlaveChangeInfo {
  required uint32 SlaveStatus = 1 [ (nanopb).int_size = IS_8 ];        // 从站状态
  required uint32 ResetCode = 2 [ (nanopb).int_size = IS_8 ];          // 开机原因
  required uint32 WorkHours = 3 [ (nanopb).int_size = IS_32 ];         // 工作时长数
  required uint32 LastHeartbeatTime = 4 [ (nanopb).int_size = IS_32 ]; // 最后一次心跳时间
}
// 查询从机请求
message QuerySlaveRequest {
  required uint32 ReqId = 1 [ (nanopb).int_size = IS_16 ]; // 请求Id
  optional FuncModuleId FMID = 2;                          // 功能模块Id
}

// 查询从机响应
message QuerySlaveResponse {
  required uint32 ReqId = 1 [ (nanopb).int_size = IS_16 ];       // 请求Id
  repeated SlaveData SlaveData = 2 [ (nanopb).max_count = 127 ]; // 从机数据
  required bool Result = 3;                                      // 查询是否成功
  required uint32 Code = 4 [ (nanopb).int_size = IS_16 ];        // 附加代码
}

// 热水箱模组配置---------WPC配置数据
message WPCConfigData {
  optional bool WPCEnable = 1;                                                // 备水功能使能
  optional uint32 RTOverflowLevelThreshold = 2 [ (nanopb).int_size = IS_16 ]; // 溢水实时阈值
  optional uint32 RTDBPLevelThreshold = 3 [ (nanopb).int_size = IS_16 ];      // 干烧实时阈值
  optional uint32 RTMidLevelThreshold = 4 [ (nanopb).int_size = IS_16 ];      // 中位实时阈值
  optional uint32 levelThreshold = 5 [ (nanopb).int_size = IS_16 ];           // 实时溢水、干烧、中位的阈值
  optional bool OutValveStatus = 6;                                           // 排废阀状态
  optional bool OverflowLevelStatus = 7;                                      // 溢水状态
  optional bool DBPLevelStatus = 8;                                           // 干烧状态
  optional bool MidLevelStatus = 9;                                           // 中位状态
  optional uint32 WPCTimeout = 10 [ (nanopb).int_size = IS_32 ];              // 备水超时时间
  optional uint32 InputWatervalue = 11 [ (nanopb).int_size = IS_16 ];         // 出热水校准-实际出水量（应该会有多个数字，比如说1000，2000，4000，5000）
  optional uint32 PulseValue = 12 [ (nanopb).int_size = IS_16 ];              // 出热水校准-实际出水量对应的脉冲数（多种出水量应该会有多种脉冲数，这边的字段需要重新设计）
  optional uint32 CAliFactor = 13 [ (nanopb).int_size = IS_16 ];              // 热水校准系数
}

// 热水箱模组配置---------WTC配置数据

  optional uint32 RtTemp = 3 [ (nanopb).int_size = IS_16 ];               // 实时温度
  optional uint32 SensorTemp = 4 [ (nanopb).int_size = IS_16 ];           // 传感器温度（小数点默认2位）
  optional uint32 MeasuredTempbByHuman = 5 [ (nanopb).int_size = IS_16 ]; // 人工测量温度（小数点默认2位）
  optional uint32 HoldTempRange = 6 [ (nanopb).int_size = IS_16 ];        // 保温温度上下限
  optional bool Reset = 7;                                                // 重置按钮状态（之前设置的温度差值数据清除用的）
}

  repeated OTAProgressData OTAProgressData = 2 [ (nanopb).max_count = 10 ];
}

message HotWaterCalibrationRspData {
  required uint32 PulseCount = 1 [ (nanopb).int_size = IS_16 ]; // 脉冲数
}

// 定量热水数据返回
message HWDCOutputRspData {
  required uint32 PulseCount = 1 [ (nanopb).int_size = IS_16 ]; // 脉冲数
}

// 任务请求报文
message TaskRequest {
  required uint32 TaskId = 1 [ (nanopb).int_size = IS_16 ];  // 任务Id
  required FuncModuleId FMID = 2;                            // 功能模块Id
  required uint32 TaskType = 3 [ (nanopb).int_size = IS_8 ]; // 任务类型（详见说明文档）
  oneof Data {
    OTAReqData OTAData = 4;
    HWDCOutputReqData HWDCOutputReqData = 5;
    HotWaterCalibrationReqData HotWaterCalibrationReqData = 6;
    TBPCReqData TBPCReqData = 7;
    BTCReqData BTCReqData = 8;
    CleanReqData CleanReqData = 9;
    FilpDoorReqData FlipDoorReqData = 10;
    TransferReqData TransferReqData = 11;
  }
}

// 任务接收结果报文
message TaskRcvResponse {
  required uint32 TaskId = 1 [ (nanopb).int_size = IS_16 ];   // 任务Id
  required uint32 TaskType = 2 [ (nanopb).int_size = IS_8 ];  // 任务类型
  required uint32 RcvResult = 3 [ (nanopb).int_size = IS_8 ]; // 任务接收结果
  required uint32 Code = 4 [ (nanopb).int_size = IS_16 ];     // 任务接收附加代码，当RcvResult为REJECTED时有意义,表示拒绝原因
}

// 任务进度报告报文
message TaskProgressReport {
  required uint32 TaskId = 1 [ (nanopb).int_size = IS_16 ];    // 任务Id
  required uint32 TaskStatus = 2 [ (nanopb).int_size = IS_8 ]; // 任务状态
  required uint32 Progress = 3 [ (nanopb).int_size = IS_8 ];   // 任务完成进度，0～100表示0%～100%
  oneof RspData {
    OTARspData TaskProgressData = 4;
  }
}

// 任务执行结果报告报文
message TaskExcResultReport {
  required uint32 TaskId = 1 [ (nanopb).int_size = IS_16 ];  // 任务Id
  required uint32 TaskType = 2 [ (nanopb).int_size = IS_8 ]; // 任务类型
  oneof RspData {
    OTARspData TaskExcResultData = 5;
    HotWaterCalibrationRspData HotWaterCalibrationRspData = 6;
    HWDCOutputRspData HWDCOutputRspData = 7;
  }
  required uint32 ExcResult = 3 [ (nanopb).int_size = IS_8 ]; // 任务执行结果
  required uint32 Code = 4 [ (nanopb).int_size = IS_16 ];     // 附加代码，当结果为ERROR和CANCEL时有意义
}

// 取消任务请求报文
message TaskCancelRequest {
  required uint32 ReqId = 1 [ (nanopb).int_size = IS_16 ];  // 请求Id
  required uint32 TaskId = 2 [ (nanopb).int_size = IS_16 ]; // 任务Id
}

// 任务取消应答报文
message TaskCancelResponse {
  required uint32 ReqId = 1 [ (nanopb).int_size = IS_16 ];  // 请求Id
  required uint32 TaskId = 2 [ (nanopb).int_size = IS_16 ]; // 任务Id
  required bool Result = 3;                                 // 取消结果
  required uint32 Code = 4 [ (nanopb).int_size = IS_16 ];   // 附加代码
}
// 查询从站运行任务报文
message QuerySlaveTaskIdRequest {
  required uint32 ReqId = 1 [ (nanopb).int_size = IS_16 ]; // 请求Id
  required FuncModuleId FMID = 2;                          // 功能模块Id
}

// 查询从站运行任务报文响应
message QuerySlaveTaskIdResponse {
  required uint32 ReqId = 1 [ (nanopb).int_size = IS_16 ];  // 请求Id
  required uint32 TaskId = 2 [ (nanopb).int_size = IS_16 ]; // 当前运行的任务ID，如果为0表示没有任务在运行
}
